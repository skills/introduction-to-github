# Prime Ingenuity Dashboard - Technical Specifications

## Overview

The Prime Ingenuity Dashboard provides real-time, transparent tracking of all collection metrics, holder analytics, staking performance, voting participation, and royalty distribution. This dashboard integrates seamlessly with the ScrollVerse ecosystem and provides comprehensive insights for collectors, stakeholders, and the community.

  lineage: [LineageEntry!]! @derivedFrom(field: "token")
  experiences: [Experience!]! @derivedFrom(field: "token")
}

type User @entity {
  id: ID!
  address: Bytes!
  tokensOwned: [Token!]! @derivedFrom(field: "currentOwner")
  tokensStaked: Int!

  votingPower: BigInt!
  votes: [Vote!]! @derivedFrom(field: "voter")

  endTime: BigInt
  duration: BigInt
}

  id: ID!

  owner: User!
  text: String!
  timestamp: BigInt!
}

  proposer: User!
  votesFor: BigInt!
  votesAgainst: BigInt!
  startTime: BigInt!
  endTime: BigInt!
  executed: Boolean!
  passed: Boolean
  votes: [Vote!]! @derivedFrom(field: "proposal")
}

type Vote @entity {
  id: ID!
  proposal: Proposal!
  voter: User!
  support: Boolean!
  votingPower: BigInt!
  timestamp: BigInt!
}

type RoyaltyDistribution @entity {
